package org.revapi.base;

import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;

import javax.annotation.Nonnull;

import org.revapi.Archive;

/**
 * A simple implementation of the {@link Archive} interface providing a file as a Revapi archive.
 *
 * @author Lukas Krejci
 * 
 * @since 0.4.1
 */
public class FileArchive implements Archive {
    private final File file;

    public FileArchive(File file) {
        this.file = file;
    }

    @Override
    public @Nonnull String getName() {
        return file.getName();
    }

    @Override
    public @Nonnull InputStream openStream() throws IOException {
        return new FileInputStream(file);
    }

    public File getFile() {
        return file;
    }
}
